实验一 表的创建、管理及数据操作

库名可以自定义成自己喜欢的,这里是MySQL的写法,其他数据库系统自行寻找别的写法

1.SQLServer

由于题目里实际只要求日期不要求时间，故我改成了date而不是继续用datetime

CREATE TABLE CP(  
    产品编号 char(6) NOT NULL,  
    产品名称 char(30) NOT NULL,  
    价格 float(8),  
    库存量 int,  
    PRIMARY KEY(产品编号)  
);  

CREATE TABLE XSS(
    客户编号 char(6) NOT NULL,
    客户名称 char(30) NOT NULL ,
    地区 char(10),
    负责人 char(8),
    电话 char(12),
    PRIMARY KEY(客户编号)
);  

CREATE TABLE CPXSB(
    产品编号 char(6) NOT NULL,
	客户编号 char(6) NOT NULL,
	销售日期 date NOT NULL,
	数量 int NOT NULL,
	销售额 float(8) NOT NULL,
	PRIMARY KEY(产品编号,客户编号,销售日期),
    FOREIGN KEY (产品编号) REFERENCES CP(产品编号) 
    ON UPDATE NO ACTION 
    ON DELETE NO ACTION,
    FOREIGN KEY (客户编号) REFERENCES XSS(客户编号) 
    ON UPDATE CASCADE 
    ON DELETE CASCADE
);

实验三 复杂查询

SELECT DISTINCT 产品名称 FROM CPXSB,CP 
WHERE CP.产品编号 = CPXSB.产品编号 
AND CONVERT(销售日期,CHAR(30)) LIKE "20%-03-18";

SELECT 产品名称,数量 FROM XSS,CPXSB,CP
WHERE XSS.客户编号 = CPXSB.客户编号 
AND XSS.客户名称='家电市场' 
AND CP.产品编号=CPXSB.产品编号 
AND CPXSB.销售日期 LIKE "20%-03-18";

3.有NULL输出

SELECT 产品名称,
IFNULL(销售日期,0) AS 销售日期,
IFNULL(客户名称,0) AS 客户名称,
IFNULL(数量,0) AS 数量
FROM CP 
LEFT OUTER JOIN CPXSB ON CP.产品编号 = CPXSB.产品编号 
LEFT OUTER JOIN XSS ON CPXSB.客户编号 = XSS.客户编号;

3.无NULL输出

SELECT CP.产品名称, CPXSB.销售日期, XSS.客户名称, CPXSB.数量
FROM CPXSB
INNER JOIN XSS
ON CPXSB.客户编号 = XSS.客户编号
INNER JOIN CP
ON CPXSB.产品编号 = CP.产品编号;

4.无NULL输出

SELECT CP.产品名称, SUM(CPXSB.数量) AS 销售总量, 
SUM(CPXSB.销售额) AS 销售总额
FROM CPXSB
INNER JOIN CP
ON CPXSB.产品编号 = CP.产品编号
GROUP BY CP.产品名称;

SELECT XSS.客户名称, SUM(CPXSB.数量) AS 购买总量, 
SUM(CPXSB.销售额) AS 购买总额
FROM CPXSB
INNER JOIN XSS
ON CPXSB.客户编号 = XSS.客户编号
GROUP BY XSS.客户名称;

SELECT 产品名称 FROM CP
WHERE CP.产品编号 NOT IN 
(SELECT DISTINCT 产品编号 FROM CPXSB 
WHERE 销售日期 LIKE "20%-03-18");

SELECT DISTINCT CP.产品名称 FROM CP
LEFT JOIN CPXSB
ON CP.产品编号 = CPXSB.产品编号 AND CPXSB.销售日期 LIKE "20%-03-18"
WHERE CPXSB.产品编号 IS NULL;

SELECT CP.产品名称 FROM CP
WHERE CP.产品编号 NOT IN (
    SELECT CPXSB.产品编号
    FROM CPXSB
    WHERE CPXSB.销售日期 LIKE "20%-03-18"
);

SELECT 产品名称,CPXSB.产品编号 FROM CP,CPXSB
WHERE CPXSB.产品编号 = CP.产品编号
AND CPXSB.数量 > ALL(
	SELECT 数量 FROM CPXSB 
	WHERE 销售日期 LIKE "20%-03-18"
);

SELECT XSS.客户名称 FROM XSS
WHERE NOT EXISTS (
    SELECT * FROM CP
    WHERE NOT EXISTS (
        SELECT * FROM CPXSB
        WHERE CPXSB.客户编号 = XSS.客户编号 
        AND CPXSB.产品编号 = CP.产品编号
    )
);

SELECT DISTINCT XSS.客户名称 FROM XSS
INNER JOIN CPXSB AS CPXSB1
ON XSS.客户编号 = CPXSB1.客户编号
WHERE NOT EXISTS (
    SELECT *
    FROM CPXSB AS CPXSB2
    WHERE CPXSB2.客户编号 = '000001' AND NOT EXISTS (
        SELECT *
        FROM CPXSB AS CPXSB3
        WHERE CPXSB3.客户编号 = XSS.客户编号 
        AND CPXSB3.产品编号 = CPXSB2.产品编号
    )
);

SELECT XSS.客户名称
FROM XSS JOIN CPXSB
ON XSS.客户编号 = CPXSB.客户编号
GROUP BY XSS.客户名称
ORDER BY SUM(CPXSB.销售额) DESC
LIMIT 1;

SELECT 产品名称 AS 销售总额前三的产品 FROM CP,CPXSB
WHERE CP.产品编号=CPXSB.产品编号
GROUP BY CPXSB.产品编号 ORDER BY SUM(销售额) DESC LIMIT 3;

SELECT CP.产品名称
FROM CP JOIN CPXSB
ON CP.产品编号 = CPXSB.产品编号
GROUP BY CP.产品名称
ORDER BY SUM(CPXSB.销售额) DESC
LIMIT 3;

讨论参照完整性
插入从表中记录时

在插入从表中记录时，如果从表中的外键约束参照了主表的主键，需要保证插入的外键值在主表的主键中存在。如果外键值不存在，就会出现参照完整性约束违规的情况。此时，可以采取以下两种方式来处理：

  1. 有限制（NO ACTION）：禁止插入从表中引用主表不存在的记录。
  2. 忽略（CASCADE）：不检查外键约束，直接插入从表记录。
删除主表中的记录时

删除主表中的记录时，如果该记录被从表引用了，则会出现参照完整性约束违规的情况。此时可以采取以下三种方式来处理：

  1. 有限制（RESTRICT）：禁止删除主表中有关联的记录，只有当从表中所有引用该主表记录的记录都被删除后，才允许删除主表中的记录。
  2. 忽略（NO ACTION或CASCADE）：不对从表中的记录做任何操作，直接删除主表中的记录。如果选择CASCADE，当主表记录被删除时，对应的从表中的记录也将被自动删除。
  3. 级联（CASCADE）：在删除主表中的记录时，同时删除所有从表中引用该主表记录的记录。
修改主表中的主键时

修改主表中的主键时，如果该主键被从表引用了，则同样会出现参照完整性约束违规的情况。此时也可以采取以上三种方式来处理。
